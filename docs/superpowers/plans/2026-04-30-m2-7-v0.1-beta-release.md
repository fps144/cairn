### Task 2: SessionLifecycleMonitorTests

**Files**: Create `Tests/CairnClaudeTests/Lifecycle/SessionLifecycleMonitorTests.swift`

测 `computeState(for:)` 纯函数(不需要 actor 启动):

**但 `computeState` 是 actor internal method**。改:暴露 internal 测试 hook `computeStateForTesting(session:)`,或者直接在测试里把 actor 方法标 nonisolated。

或者更简:暴露 **5 个场景** 的静态纯函数版:

```swift
extension SessionLifecycleMonitor {
    /// T2 测试用 —— 纯函数版(不依赖 actor state)
    /// mtimeAge:文件 mtime 距今秒数;fileExists:文件是否存在
    /// hangingToolUses:未配对 tool_use 数
    nonisolated static func computeState(
        mtimeAge: TimeInterval,
        fileExists: Bool,
        hangingToolUses: Int
    ) -> SessionState {
        if !fileExists { return .crashed }
        if mtimeAge < 60 { return .live }
        if mtimeAge < 5 * 60 { return .idle }
        if hangingToolUses == 0 { return .ended }
        if mtimeAge >= 30 * 60 { return .abandoned }
        return .idle
    }
}
```

原 instance method 改调此纯函数,保持行为一致。

测试:
```swift
func test_live() {
    XCTAssertEqual(
        SessionLifecycleMonitor.computeState(mtimeAge: 30, fileExists: true, hangingToolUses: 0),
        .live
    )
}
func test_idle() { /* mtimeAge=120 → .idle */ }
func test_ended_noHanging() { /* mtimeAge=600 (10min), hanging=0 → .ended */ }
func test_abandoned_longWithHanging() { /* mtimeAge=1800 (30min), hanging=1 → .abandoned */ }
func test_idle_pendingButShort() { /* mtimeAge=600, hanging=1 → .idle(not yet abandoned) */ }
func test_crashed_fileMissing() { /* fileExists=false → .crashed */ }
```

6 测试,覆盖 5 态 + 文件缺失。

### Task 6: build-release-dmg.sh

**Files**: Create `scripts/build-release-dmg.sh`

```bash
#!/bin/bash
# build-release-dmg.sh — release 构建 + app bundle + 未签名 DMG
#
# 产物:dist/Cairn-v0.1.0-beta-macos-arm64.dmg
#
# 依赖:
#   - macOS 14+
#   - Xcode 15+(swift toolchain)
#   - hdiutil(macOS 原生)
#
# 不签名、不公证。首次运行需 `xattr -rd com.apple.quarantine /Applications/Cairn.app`。

set -euo pipefail

ROOT_DIR="$(cd "$(dirname "$0")/.." && pwd)"
cd "$ROOT_DIR"

# v0.1 Beta 只支持 arm64(spec §8.5)。Intel Mac 上 build 会产 x86_64 bin
# 但 DMG 名字会误导。拒绝在非 arm64 上跑。
ARCH=$(uname -m)
if [[ "$ARCH" != "arm64" ]]; then
    echo "[build-dmg] ERROR: v0.1 Beta 只支持 Apple Silicon (arm64),当前 $ARCH" >&2
    echo "[build-dmg]   Intel Universal Binary 留 v0.2" >&2
    exit 1
fi

VERSION="${CAIRN_VERSION:-0.1.0-beta}"
DMG_NAME="Cairn-v${VERSION}-macos-arm64"
DIST_DIR="dist"
BUILD_DIR="build"
STAGING_DIR="$BUILD_DIR/dmg-staging"

echo "[build-dmg] 1/5 Release build ..."
./scripts/make-app-bundle.sh release

if [[ ! -d "$BUILD_DIR/Cairn.app" ]]; then
    echo "[build-dmg] ERROR: $BUILD_DIR/Cairn.app 不存在" >&2
    exit 1
fi

echo "[build-dmg] 2/5 准备 staging 目录 ..."
rm -rf "$STAGING_DIR"
mkdir -p "$STAGING_DIR"
cp -R "$BUILD_DIR/Cairn.app" "$STAGING_DIR/Cairn.app"
ln -s /Applications "$STAGING_DIR/Applications"

# 加 INSTALL.txt 提示 xattr
cat > "$STAGING_DIR/INSTALL.txt" <<'EOF'
Cairn v0.1 Beta — 安装说明

1. 把 Cairn.app 拖到右侧 Applications 文件夹
2. 首次运行前,打开 Terminal.app 执行:

       sudo xattr -rd com.apple.quarantine /Applications/Cairn.app

3. 正常双击打开 Cairn.app

(Cairn 走永不签名路线 — MIT 项目不购买 Apple Developer 账号)

更多见 README:https://github.com/fps144/cairn
EOF

echo "[build-dmg] 3/5 生成 DMG ..."
mkdir -p "$DIST_DIR"
DMG_PATH="$DIST_DIR/${DMG_NAME}.dmg"
rm -f "$DMG_PATH"

hdiutil create \
    -volname "Cairn" \
    -srcfolder "$STAGING_DIR" \
    -ov \
    -format UDZO \
    "$DMG_PATH"

echo "[build-dmg] 4/5 计算 SHA256 ..."
SHA256=$(shasum -a 256 "$DMG_PATH" | awk '{print $1}')
echo "$SHA256  ${DMG_NAME}.dmg" > "$DIST_DIR/${DMG_NAME}.sha256"

echo "[build-dmg] 5/5 产物:"
ls -la "$DIST_DIR"
echo ""
echo "DMG: $DMG_PATH"
echo "SHA256: $SHA256"
```
